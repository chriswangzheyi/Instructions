# 准备数据


```python
import numpy as np
import matplotlib.pyplot as plt
```


```python
w,b=1.8,2.5
```


```python
np.random.seed(0)  # 定义随机数种子
x = np.random.rand(100)*4-2 # 生成100个-2到2的随机数
noise = np.random.randn(100)/3 # 生成100个随机噪声
y = w*x +b + noise

```


```python
x = x.reshape(-1,1) # 从一行变为一列
```


```python
x.shape, y.shape
```




    ((100, 1), (100,))




```python
plt.scatter(x,y)
plt.show()
```

# 训练模型


```python
from sklearn.neural_network import MLPRegressor
```


```python
reg = MLPRegressor(
    hidden_layer_sizes=(1,),
    activation='identity', #不加激活函数
    learning_rate_init=0.01, # 学习率
    random_state=233
)
```


```python
reg.fit(x,y)
```
