### 代码实现

	import numpy as np
	import matplotlib.pyplot as plt
	
	x = np.linspace(-6,4,100)
	y = x**2+2*x+3
	
	theta = 3
	alpha = 0.06
	iterationNum = 100
	
	fig, ax =plt.subplots()
	ax.plot(x,y)
	for i in range(iterationNum):
	    ax.scatter(theta, theta**2+2*theta+3)
	    ax.annotate(i,(theta,theta**2+2*theta+3))
	    theta = theta - alpha * (2*theta +2)
	
	print(theta)
	plt.show()
