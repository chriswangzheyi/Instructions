### 创建一个向量
	
	x = [[1.]]
	m = tf.matmul(x,x)
	print(m)

结果

	tf.Tensor([[1.]], shape=(1, 1), dtype=float32)
	
	
### 构建一个矩阵
	x = tf.constant([[1,9],[3,6]])
	print(x)

结果

	tf.Tensor(
	[[1 9]
	 [3 6]], shape=(2, 2), dtype=int32)
	 
	 
### 转化格式

	y=x.numpy()
	print(y) 	 
	
结果

	[[1 9]
	 [3 6]]
