	def binartSearch(self, nums, target):
        # Write your code here.
        start , end = 0 , len(nums) - 1

        while start + 1< end:
            mid = (start + end) // 2

            if nums[mid] < target:
                start = mid
            elif nums[mid] == target:
                end = mid
            else:
                end = mid
        
        if nums[start] == target:
            return start
        if nums[end] == target:
            return end
        
        return -1
