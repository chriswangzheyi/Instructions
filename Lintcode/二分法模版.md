	def binartSearch(self, nums, target):
		if not nums:
			return -1
		
		while start + 1 < end:
			# 避免死循环
			mid = (start + end ) // 2
			
			if nums [mid] < target:
				start = mid
			
			elif nums[mid] == target:
				end = mid
				
			else:
				end = mid
			
		if nums [start] == target:
			return start
		if nums [end] == target:
			return end
		
		return -1
